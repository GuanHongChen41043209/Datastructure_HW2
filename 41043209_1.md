## AdjacencyMatrixGraph
```cpp
class WeightedAdjacencyMatrixGraph {
private:
    int n; 
    int e; 
    vector<vector<int>> matrix;
};
```
## 初始化AdjacencyMatrix
```cpp
WeightedAdjacencyMatrixGraph(int vertices) {
    n = vertices;
    e = 0;
    matrix.resize(n, vector<int>(n, 0));
}
```
## 判斷是否是空的
```cpp
bool IsEmpty() const {
    return n == 0;
}
```
## 回傳有幾個點
```cpp
int NumberOfVertices() const {
    return n;
}
```
## 回傳有幾個邊
```cpp
int NumberOfEdges() const {
    return e;
}

## 新增邊
```cpp
void InsertEdge(int u, int v, int weight) {
    if (u < 0 || u >= n || v < 0 || v >= n) {
        cout << "Invalid vertex!" << endl;
        return;
    }
    if (weight <= 0) {
        cout << "Weight must be positive!" << endl;
        return;
    }
    if (matrix[u][v] == 0) {
        e++;
    }
    matrix[u][v] = weight;
    matrix[v][u] = weight;
}
```
## 回傳這個邊是否存在
```cpp
bool ExistsEdge(int u, int v) const {
    if (u < 0 || u >= n || v < 0 || v >= n) {
    return false;
    }
    return matrix[u][v] != 0;
}
```
## 計算點的degree
```cpp
int Degree(int u) const {
    if (u < 0 || u >= n) {
        cout << "Invalid vertex!" << endl;
        return -1;
    }
    int count = 0;
    for (int v = 0; v < n; v++) {
        if (matrix[u][v] != 0) {
            count++;
        }
    }
    return count;
}
```
## 刪除點
```cpp
void DeleteVertex(int v) {
    if (v < 0 || v >= n) {
        cout << "Invalid vertex!" << endl;
        return;
    }
    for (int i = 0; i < n; i++) {
        if (matrix[v][i] != 0) {
            e--;
        }
    }
    matrix.erase(matrix.begin() + v);
    for (int i = 0; i < n - 1; i++) {
        matrix[i].erase(matrix[i].begin() + v);
    }
    n--;
}
```
## 輸出Adjacency Matrix
```cpp
void PrintMatrix() const {
    cout << "Adjacency Matrix:" << endl;
    for (int i = 0; i < n; i++) {
        for (int j = 0; j < n; j++) {
            cout << setw(5) << matrix[i][j];
        }
        cout << endl;
    }
}
```
